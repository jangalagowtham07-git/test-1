Action()
{ 
	char *city;
	int pincode;
	char *tool;
	
	lr_save_string("banglore","new city");
	lr_save_int(560037,"areapincode");
	lr_save_string("welcome to loadrunner","PTtool");
	
	city=lr_eval_string("{new city}");
	pincode=atoi(lr_eval_string("{areapincode}"));
	tool=lr_eval_string("{PTtool}");
	
	lr_output_message("%s",city);
	lr_output_message("%d",pincode);
	lr_output_message("%s",tool);
		
	return 0;
}
